1. 3(x + 5)

2. -4(y - 10)

3. (2m + 5)(3m - 2)

Solution

  • = 2m(3m - 2) + 5(3m - 2)
  • = 6m² - 4m + 15m - 10
  • = 6m² + 11m - 10

4. 7(p + 3)(2p - 4)

Solution

  • = (7p + 21)(2p - 4)
  • = 7p(2p - 4) + 21(2p - 4)
  • = 14p² - 28p + 42p - 84
  • = 14p² + 14p - 84

5. -5(3x - 1)(5x - 2) - (x + 2)²

  • = (-15x + 5)(5x - 2) - (x + 2)²
  • = -75x² + 30x + 25x - 10 - (x + 2)(x + 2)
  • = -75x² + 55x - 10 - (x² + 2x + 2x + 4)
  • = -75x² + 55x - 10 - (x² + 4x + 4)
  • = -75x² + 55x - 10 - x² - 4x - 4
  • = -76x² + 51x - 14

b 4(x + 2) - x(x + 2)

Solution

c y² - 6y - 27

  • = y² - 9y + 3y - 27
  • = (y² - 9y) + (3y - 27)
  • = y(y - 9) + 3(y - 9)
  • = (y + 3)(y - 9)

d 2d² - 22d + 56

Solution

  • = 2d² - 8d - 14d + 56
  • = (2d² - 8d) - (14d - 56)
  • = 2d(d - 4) - 14(d - 4)
  • = (2d - 14)(d - 4)

e bx² + 10bx - 24b

  • = bx² + 12bx - 2bx - 24b
  • = (bx² + 12bx) - (2bx + 24b)
  • = bx(x + 12) - 2b(x + 12)
  • = (bx - 2b)(x + 12)
